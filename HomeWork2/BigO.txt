Реализуйте и оцените алгоритм поиска максимального числа в 
неотсортированном массиве (объясните почему именно такая оценка). 
Реализовать можно как в виде текста, псевдокода или блок-схемы, так и в виде реального кода.


function findMax(arr){
        let max = arr[0];       // 1
        for (let i = 1; i < arr.length; i++) {   
            if (arr[i]>max){     //  n
                max = arr[i];
            }
          }
        return max;    
    }


O(1+n) = O(n)
